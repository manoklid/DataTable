// Function to simulate API call with delay
function fetchDataFromAPI() {
    return new Promise((resolve, reject) => {
        // Simulate network delay
        setTimeout(() => {
            // Simulate random API failure (10% chance)
            if (Math.random() < 0.1) {
                reject("API server unavailable. Please try again later.");
            } else {
                resolve(employeeData);
            }
        }, 1000); // 1 second delay
    });
}

// Function to load data into the table
function loadData() {
    // Show loading state
    $('#employeeTable tbody').html('<tr><td colspan="8" class="loading"><i class="fas fa-spinner fa-spin"></i> Loading data from API...</td></tr>');
    $('#errorMessage').hide();
    
    // Fetch data from simulated API
    fetchDataFromAPI()
        .then(data => {
            // Clear the table
            if ($.fn.DataTable.isDataTable('#employeeTable')) {
                $('#employeeTable').DataTable().destroy();
            }
            $('#employeeTable tbody').empty();
            
            // Initialize DataTable with columnDefs
            $('#employeeTable').DataTable({
                data: data,
                pageLength: 5,
                lengthMenu: [[5, 10, 25, 50, -1], [5, 10, 25, 50, "All"]],
                responsive: true,
                columnDefs: [
                    // ID column
                    { 
                        targets: 0,
                        render: function(data, type, row) {
                            return '<span style="font-weight: bold;">' + data + '</span>';
                        }
                    },
                    // Name column - make rows 1, 2, and 4 links (0-indexed 0, 1, 3)
                    { 
                        targets: 1,
                        render: function(data, type, row, meta) {
                            if (meta.row === 0 || meta.row === 1 || meta.row === 3) {
                                return '<a href="' + row.profileUrl + '" target="_blank"><i class="fas fa-user-circle"></i> ' + data + '</a>';
                            }
                            return '<i class="fas fa-user"></i> ' + data;
                        }
                    },
                    // Department column
                    { 
                        targets: 2,
                        render: function(data, type, row) {
                            return '<span class="department department-' + data.toLowerCase().replace(/\s+/g, '-') + '">' + data + '</span>';
                        }
                    },
                    // Email column
                    { 
                        targets: 3,
                        render: function(data, type, row) {
                            return '<a href="mailto:' + data + '"><i class="fas fa-envelope"></i> ' + data + '</a>';
                        }
                    },
                    // Status column
                    { 
                        targets: 4,
                        render: function(data, type, row) {
                            const statusClass = data === 'Active' ? 'status-active' : 'status-leave';
                            return '<span class="' + statusClass + '"><i class="fas fa-circle"></i> ' + data + '</span>';
                        }
                    },
                    // Salary column
                    { 
                        targets: 5,
                        render: function(data, type, row) {
                            return '$' + data.toLocaleString();
                        }
                    },
                    // Join Date column
                    { 
                        targets: 6,
                        render: function(data, type, row) {
                            return new Date(data).toLocaleDateString('en-US', { 
                                year: 'numeric', 
                                month: 'short', 
                                day: 'numeric' 
                            });
                        }
                    },
                    // Actions column
                    { 
                        targets: 7,
                        orderable: false,
                        render: function(data, type, row) {
                            return '<button style="padding: 5px 10px; margin-right: 5px;"><i class="fas fa-eye"></i></button>' +
                                   '<button style="padding: 5px 10px;"><i class="fas fa-edit"></i></button>';
                        }
                    }
                ],
                columns: [
                    { data: 'id', title: 'ID' },
                    { data: 'name', title: 'Name' },
                    { data: 'department', title: 'Department' },
                    { data: 'email', title: 'Email' },
                    { data: 'status', title: 'Status' },
                    { data: 'salary', title: 'Salary' },
                    { data: 'joinDate', title: 'Join Date' },
                    { data: null, title: 'Actions' }
                ]
            });
        })
        .catch(error => {
            $('#errorText').text(error);
            $('#errorMessage').show();
            $('#employeeTable tbody').html('<tr><td colspan="8" class="loading">Failed to load data. Click "Load Data" to try again.</td></tr>');
        });
}

// Event handlers
$(document).ready(function() {
    // Load data when page is ready
    loadData();
    
    // Add event listeners to buttons
    $('#loadDataBtn').on('click', function() {
        loadData();
    });
    
    $('#refreshBtn').on('click', function() {
        location.reload();
    });
});
